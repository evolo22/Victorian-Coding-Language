Program
    ::= Header Body Signature

Header
    ::= "Dear" FileName "," ImportList?

ImportList
    ::= Identifier ("," Identifier)*

Body
    ::= (GlobalDecl | FunctionDecl)*

Signature
    ::= "Signed," MainDesignation?

MainDesignation
    ::= "Designated principle function" Identifier


GlobalDecl
    ::= "Highly Esteemed" VarDecl


FunctionDecl
    ::= "Do" Identifier "(" ParamList? ")" Block

ParamList
    ::= Identifier ("," Identifier)*


Block
    ::= "{" Statement* "}"


Statement
    ::= VarDecl
     | Assignment
     | IfStmt
     | WhileStmt
     | ForStmt
     | PrintStmt
     | ReturnStmt
     | ExprStmt


VarDecl
    ::= ("Madame" | "Lord") Identifier ("=" Expr)?


Assignment
    ::= Identifier "=" Expr


PrintStmt
    ::= "Inscribe" "(" Expr ")"


ReturnStmt
    ::= "Hasten back" Expr


IfStmt
    ::= "If" "(" Condition ")" "is so" Block
        ("Otherwise if" "(" Condition ")" "is so" Block)*
        ("Otherwise" Block)?
     |  "If" "(" Condition ")" "is not so" Block


WhileStmt
    ::= "Whilst" "(" Condition ")" "persists" Block


ForStmt
    ::= "For each" Identifier "contained in" Identifier Block
     |  "For each" Identifier "from" Expr "to" Expr Block


ExprStmt
    ::= Expr


Expr
    ::= Term (("+" | "-") Term)*

Term
    ::= Factor (("*" | "/") Factor)*

Factor
    ::= Number
     | String
     | Identifier
     | FunctionCall
     | "(" Expr ")"


FunctionCall
    ::= Identifier "(" ArgList? ")"

ArgList
    ::= Expr ("," Expr)*


Condition
    ::= Expr Comparator Expr


Comparator
    ::= "of greater magnitude than"
     |  "of greater or equal magnitude than"
     |  "of lesser magnitude than"
     |  "of lesser or equal magnitude than"
     |  "of equal magnitude to"
     |  "of unequal magnitude to"


Type
    ::= "Note"
     |  "Digit"
     |  "Numeral"
     |  "Palter"
     |  "BasketOfOranges"
     |  "Lexicon"
     |  "CoffeeSisters"


Identifier
    ::= Letter (Letter | Digit | "_")*

Number
    ::= Digit+

String
    ::= "\"" (Character)* "\""


FileName
    ::= Identifier "." "vic"
